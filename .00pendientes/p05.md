El fallo viene de que el store `web_frontend/stores/auth.ts` está usando `useCookie()` directamente dentro de `hydrateFromCookie()` y `setToken()`, y además `bootstrap()` termina llamando `logout()` cuando falla `fetchMe()`. Ese patrón hace que, en SSR o después de ciertos `await`, Nuxt ya no tenga contexto activo y reviente exactamente como en el stack trace. Eso sí se ve en el store actual .
También se ve que el flujo actual llama `authStore.setToken(...)` desde login/registro y `authStore.logout()` desde los layouts, por lo que el problema no está aislado a una sola pantalla .

La corrección estable es esta:

* el **store** deja de tocar cookies;
* el **composable `useAuth()`** pasa a ser el único dueño de la cookie;
* login, registro y logout sincronizan cookie desde `useAuth()`;
* `bootstrapAuth()` lee la cookie con contexto Nuxt válido y se la pasa al store.
  Con eso, el tema queda cerrado de forma limpia y no debería volver a romperse por SSR.

---

# Impacto y riesgos

No se toca backend.
No se toca base de datos.
Solo se corrige el manejo de sesión del frontend Nuxt.

Riesgo evitado:

* volver a llamar composables Nuxt desde Pinia fuera de contexto;
* romper logout/login en SSR;
* dejar cookies y estado desincronizados.

```ts id="mywbrl"
import { FetchError } from 'ofetch'
import { defineStore } from 'pinia'

import type { CurrentUser, RoleCode } from '~/types/auth'

type AuthState = {
  token: string | null
  user: CurrentUser | null
  loaded: boolean
}

function getRoleCodes(user: CurrentUser | null): RoleCode[] {
  if (!user) {
    return []
  }

  if (user.role_codes?.length) {
    return user.role_codes
  }

  return user.roles?.map(role => role.code) ?? []
}

export const useAuthStore = defineStore('auth', {
  state: (): AuthState => ({
    token: null,
    user: null,
    loaded: false,
  }),

  getters: {
    roleCodes: (state): RoleCode[] => getRoleCodes(state.user),
    isAuthenticated: (state): boolean => Boolean(state.token),
  },

  actions: {
    hydrate(token: string | null) {
      this.token = token
      return this.token
    },

    setToken(token: string | null) {
      this.token = token
    },

    async fetchMe() {
      const { apiFetch } = useApi()

      try {
        this.user = await apiFetch<CurrentUser>('/auth/me', { method: 'GET' })
      }
      catch (error: unknown) {
        if (error instanceof FetchError && error.statusCode === 404) {
          this.user = await apiFetch<CurrentUser>('/users/me', { method: 'GET' })
        }
        else {
          this.loaded = true
          throw error
        }
      }

      this.loaded = true
      return this.user
    },

    async bootstrap(initialToken: string | null = null) {
      this.hydrate(initialToken)

      if (!this.token) {
        this.user = null
        this.loaded = true
        return null
      }

      try {
        return await this.fetchMe()
      }
      catch {
        this.user = null
        this.token = null
        this.loaded = true
        return null
      }
    },

    logout() {
      this.user = null
      this.token = null
      this.loaded = true
    },
  },
})
```

```ts id="g4p9k6"
import type { CurrentUser, RoleCode } from '~/types/auth'

const adminRoleCodes: RoleCode[] = [
  'super_admin',
  'admin_validation',
  'admin_support',
  'admin_moderation',
  'admin_privacy',
]

export function useAuth() {
  function getRoleCodes(user: CurrentUser | null | undefined): RoleCode[] {
    if (!user) {
      return []
    }

    if (user.role_codes?.length) {
      return user.role_codes
    }

    return user.roles?.map(role => role.code) ?? []
  }

  function isAdminRole(role: RoleCode): boolean {
    return adminRoleCodes.includes(role)
  }

  function resolveRoleHome(roles: RoleCode[]): string {
    if (roles.includes('privacy_auditor')) {
      return '/privacy-auditor/access-logs'
    }

    if (roles.some(isAdminRole)) {
      return '/admin/dashboard'
    }

    if (roles.includes('professional')) {
      return '/professional/dashboard'
    }

    return '/patient/dashboard'
  }

  function syncAccessToken(token: string | null) {
    const cookie = useCookie<string | null>('access_token')
    cookie.value = token
  }

  async function bootstrapAuth() {
    const authStore = useAuthStore()
    const cookie = useCookie<string | null>('access_token')
    const cookieToken = cookie.value ?? null

    if (!authStore.loaded) {
      await authStore.bootstrap(cookieToken)
      return authStore
    }

    if (!authStore.token && cookieToken) {
      authStore.setToken(cookieToken)
    }

    return authStore
  }

  return {
    adminRoleCodes,
    bootstrapAuth,
    getRoleCodes,
    isAdminRole,
    resolveRoleHome,
    syncAccessToken,
  }
}
```
